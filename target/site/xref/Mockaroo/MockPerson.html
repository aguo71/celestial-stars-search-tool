3   /**
4    * Class representing a mock entry in a csv.
5    */
6   public class MockPerson {
7     private String fname;
8     private String lname;
9     private String time;
10    private String email;
11    private String gender;
12    private String address;
13  
14    /**
15     *
16     * @param time time of entry
17     * @param fname first name
18     * @param lname last name
19     * @param email email
20     * @param gender gender
21     * @param address home address
22     */
23    public MockPerson(String time, String fname, String lname, String email, String gender,
24                      String address) {
25      this.fname = fname;
26      this.lname = lname;
27      this.time = time;
28      this.email = email;
29      this.gender = gender;
30      this.address = address;
31    }
32  
33    @Override
34    public String toString() {
35      String toRet = "";
36      if (!time.equals("")) {
37        toRet = toRet + "Entry from: " + time + ". ";
38      }
39      if (!fname.equals("") && lname.equals("")) {
40        toRet = toRet + "Name: " + fname + ". ";
41      } else if (!fname.equals("")) {
42        toRet = toRet + "Name: " + fname + " " + lname + ". ";
43      } else {
44        if (!lname.equals("")) {
45          toRet = toRet + "Name: " + lname + ". ";
46        }
47      }
48      if (!gender.equals("")) {
49        toRet = toRet + "Gender: " + gender + ". ";
50      }
51      if (!email.equals("")) {
52        toRet = toRet + "Email: " + email + ". ";
53      }
54      if (!address.equals("")) {
55        toRet = toRet + "Home Address: " + address + ".";
56      }
57      return toRet;
58    }
59  }
