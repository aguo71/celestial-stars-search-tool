14  public class Repl {
15      Map<String, Action> commands = new HashMap<>();
16  
17      public Repl() {}
18  
19      public void registerAction(String method, Action action) {
20          commands.put(method, action);
21      }
22  
23      public void readCommands() {
24          BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
25          String input;
26          while(true) {
27              try {
28                  input = reader.readLine();
29              } catch (IOException e) {
30                  System.out.println("ERROR: IOException occurred during REPL reading");
31                  continue;
32              }
33              if(input == null) {
34                  return;
35              }
36  
37              // parses REPL command, splitting on spaces except within double quotes
38              // https://stackoverflow.com/questions/366202/regex-for-splitting-a-string-using-space-when-not-surrounded-by-single-or-double
39  
40              List<String> matchList = new ArrayList<>();
41              Pattern regex = Pattern.compile("[^\\s\"']+|\"[^\"]*\"|'[^']*'");
42              Matcher regexMatcher = regex.matcher(input);
43              while (regexMatcher.find()) {
44                  matchList.add(regexMatcher.group());
45              }
46              String[] tokens = new String[matchList.size()];
47              tokens = matchList.toArray(tokens);
48  
49              if(commands.containsKey(tokens[0])) {
50                  commands.get(tokens[0]).run(tokens);
51              } else {
52                  System.out.println("ERROR: Invalid command");
53              }
54          }
55      }
56  }
