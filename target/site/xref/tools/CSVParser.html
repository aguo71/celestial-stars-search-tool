3   import java.io.BufferedReader;
4   import java.io.IOException;
5   
6   /**
7    * Class that parses CSV files.
8    */
9   public class CSVParser {
10    /**
11     * Empty constructor.
12     */
13    public CSVParser() { }
14  
15    /**
16     * Parses a single CSV line.
17     * @param reader BufferedReader reading from desired csv file
18     * @param delims delims to tokenize each line read with
19     * @return array of string tokens in the line read
20     * @throws IOException if readLine fails
21     */
22    public String[] parseLine(BufferedReader reader, String delims) throws IOException {
23      String line = reader.readLine();
24      if (line == null) {
25        return null;
26      } else {
27        // Tokenizes line based on deliminators, keeping all tokens even if empty
28        return line.split(delims, -1);
29      }
30    }
31  }
