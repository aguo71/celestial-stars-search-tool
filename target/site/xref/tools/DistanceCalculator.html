3   import java.util.List;

5   /**
6    * Class for a method to find distance between two sets of coordinates.
7    */
8   public class DistanceCalculator {
9     /**
10     * Empty constructor.
11     */
12    public DistanceCalculator() { }
13  
14    /**
15     * Finds Euclidean distance of 2 sets of coordinates.
16     * @param c1 coordinate 1
17     * @param c2 coordinate 2
18     * @return distance between input coordinates
19     */
20    public double getDistance(List<Double> c1, List<Double> c2) {
21      if (c1.size() != c2.size()) {
22        throw new RuntimeException("ERROR: Can't calculate distance");
23      }
24  
25      double dist = 0;
26      for (int x = 0; x < c1.size(); x++) {
27        dist += Math.pow(c1.get(x) - c2.get(x), 2);
28      }
29      return Math.sqrt(dist);
30    }
